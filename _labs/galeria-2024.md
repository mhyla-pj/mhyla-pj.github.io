<script>
  (function () {
    var lightbox = document.getElementById('holiday-lightbox');
    if (!lightbox) return;

    var image = lightbox.querySelector('.holiday-lightbox__image');
    var triggers = Array.prototype.slice.call(document.querySelectorAll('.holiday-card__button'));
    var counter = lightbox.querySelector('.holiday-lightbox__counter');
    var activeIndex = -1;

    var updateImage = function (index) {
      var trigger = triggers[index];
      if (!trigger) return;
      var img = trigger.querySelector('img');
      if (!img) return;
      image.src = trigger.dataset.full || img.src;
      image.alt = img.alt || 'Kartka swiateczna';
      activeIndex = index;
      if (counter) counter.textContent = (index + 1) + ' / ' + triggers.length;
    };

    var openLightbox = function (index) {
      updateImage(index);
      lightbox.classList.add('is-open');
      lightbox.setAttribute('aria-hidden', 'false');
      document.body.classList.add('holiday-lightbox-open');
    };

    var closeLightbox = function () {
      lightbox.classList.remove('is-open');
      lightbox.setAttribute('aria-hidden', 'true');
      image.src = '';
      image.alt = '';
      document.body.classList.remove('holiday-lightbox-open');
    };

    var moveBy = function (direction) {
      if (activeIndex < 0) return;
      var nextIndex = (activeIndex + direction + triggers.length) % triggers.length;
      updateImage(nextIndex);
    };

    triggers.forEach(function (trigger, index) {
      trigger.addEventListener('click', function () {
        openLightbox(index);
      });
    });

    lightbox.addEventListener('click', function (event) {
      if (!event.target || !event.target.dataset) return;
      if (event.target.dataset.close === 'true') {
        closeLightbox();
        return;
      }
      if (event.target.dataset.nav === 'prev') {
        moveBy(-1);
      }
      if (event.target.dataset.nav === 'next') {
        moveBy(1);
      }
    });

    document.addEventListener('keydown', function (event) {
      if (!lightbox.classList.contains('is-open')) return;
      if (event.key === 'Escape') closeLightbox();
      if (event.key === 'ArrowLeft') moveBy(-1);
      if (event.key === 'ArrowRight') moveBy(1);
    });
  })();
</script>
